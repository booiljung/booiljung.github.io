15.4.4 JSON/JSON5 기반 ACL 설정 파일 작성 문법 및 예제

말로만 백날 설명하는 것은 끝났다. 이제 데몬(Routers)을 틀어막는 가장 기초적이며 파괴적인 JSON 검열 룰북을 완성하라.

0.1 [Runbook] 실전 차단(Kill-Switch) 코드 구현

// zenohd.json5
plugins: {
  access_control: {
    // 1. 유저를 역할(Role)에 매핑한다. (인증 플러그인에서 넘어온 유저 대상)
    users: {
      "drone_01": "worker",
      "drone_02": "worker",
      "monitoring_server": "inspector",
      "admin_kim": "god_mode"
    },

    // 2. 역할별 권한(ACL)을 강제한다.
    roles: {
      "worker": {
        // [허용 리스트] 센서값을 쓰고(Put), 다른 놈들 데이터를 구독(Sub)할 수 있다.
        allow: [
          "Put:drone/telemetry/**",
          "Sub:drone/telemetry/**"
        ],
        // [금지 리스트] 무슨 일이 있어도 루트 패킷 전체를 빨아먹거나 지우지 못하게 죽여라.
        deny: [
          "Query:**",
          "Delete:**",
          "Put:system/config/**"
        ]
      },

      "inspector": {
        // 대시보드 관제탑은 오직 과거 기록(쿼리)과 구독(Sub)만 된다. 쓰는 건 불가능하다.
        allow: ["Query:**", "Sub:**"]
      },

      "god_mode": {
        // 모든 권한 다 열어줌
        allow: ["**"]
      }
    }
  }
}

이 플러그인이 켜지는 순간, Zenoh 라우터는 데이터가 목적지에 도달하기도 전에, 혹은 백엔드 DB 에 닿기도 전에 소켓 계층에서 저 deny 블랙리스트 룰에 걸린 모든 패킷을 빛의 속도로 묵살(Drop) 해버린다.

9.4.1.2 C 구조체(Struct)와 Go 구조체 간의 데이터 패킹 및 언패킹 전략

Protobuf도 무겁다고 느껴지는가? 외부 라이브러리(헤더 파일) 한 줄 의존하는 것조차 싫다면 가장 원초적인 방법이 있다. C 구조체의 포인터를 (uint8_t*) 로 캐스팅한 후 통째로 발사하고, Go 에서 unsafe 패키지를 이용해 그 메모리 덩어리에 강제로 Go 구조체의 껍데기를 씌워버리는 빙의(Casting) 마법이다.

C 언어의 발사 (Struct 덤프)

// 반드시 패딩(Padding)을 없애서 순수 바이트 일렬종대로 세워야 한다!
#pragma pack(push, 1)
typedef struct {
    uint32_t id;
    float temperature;
} RawSensorData;
#pragma pack(pop)

RawSensorData data = { .id = 404, .temperature = 98.6f };

// 내 구조체가 박힌 메모리의 시작 주소(&data)부터 사이즈(sizeof)만큼 전송!!
z_publisher_put(z_loan(pub), (const uint8_t*)&data, sizeof(RawSensorData), NULL);

Go 언어의 빙의 수신 (Unsafe 패싱)

package main

import (
	"bytes"
	"encoding/binary"
	"fmt"
	"github.com/eclipse-zenoh/zenoh-go"
)

// C와 정확히 100% 동일한 필드 순서와 크기로 배열해야 한다.
type GoRawSensorData struct {
	ID          uint32 // 4바이트
	Temperature float32 // 4바이트
}

func parseStructCallback(sample zenoh.Sample) {
	buf := bytes.NewReader(sample.Payload())
	var decodedData GoRawSensorData
	
	// [핵심] 리틀 엔디안(Little Endian) 규칙에 따라 바이너리를 Go 구조체 슬롯에 강제 조립한다.
	err := binary.Read(buf, binary.LittleEndian, &decodedData)
	if err != nil {
		fmt.Println("파싱 실패 (바이트열 크기가 구조체와 불일치!)")
		return
	}
	
	fmt.Printf("아이디: %d, 온도: %f\n", decodedData.ID, decodedData.Temperature)
}

이 기법은 컴파일 시점에 CPU 의 엔디안(Endianness) 정보나 타입 정렬(Alignment) 이 조금이라도 엇나가면 “온도 98.6” 이 “온도 14,024,000,000” 으로 둔갑하여 읽히는 대재앙을 낳는다.
속도는 0.0001ms 에 수렴할 만큼 빠르지만, 철저한 바이트 규약 테스트가 선행되어야만 쓸 수 있는 마검(魔劍)이다.

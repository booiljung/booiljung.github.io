659.25 TransformBroadcasterмқҳ rclpy мӮ¬мҡ©лІ•

1. к°ңмҡ”

TransformBroadcasterлҠ” tf2_ros нҢЁнӮӨм§Җм—җм„ң м ңкіөн•ҳлҠ” н•өмӢ¬ нҒҙлһҳмҠӨлЎңм„ң, лҸҷм Ғ мўҢн‘ң ліҖнҷҳ(dynamic transform)мқ„ /tf нҶ н”ҪмңјлЎң л°ңн–үн•ҳлҠ” кё°лҠҘмқ„ лӢҙлӢ№н•ңлӢӨ. Python нҒҙлқјмқҙм–ёнҠё лқјмқҙлёҢлҹ¬лҰ¬мқё rclpyлҘј нҶөн•ҳм—¬ TransformBroadcasterлҘј нҷңмҡ©н•ҳл©ҙ, лЎңлҙҮ мӢңмҠӨн…ң лӮҙм—җм„ң мӢңк°„м—җ л”°лқј ліҖнҷ”н•ҳлҠ” мўҢн‘ң н”„л Ҳмһ„ к°„мқҳ кҙҖкі„лҘј мӢӨмӢңк°„мңјлЎң л°ңн–үн•  мҲҳ мһҲлӢӨ. ліё м Ҳм—җм„ңлҠ” tf2_ros.TransformBroadcasterмқҳ rclpy кё°л°ҳ мӮ¬мҡ©лІ•мқ„ мІҙкі„м ҒмңјлЎң кё°мҲ н•ңлӢӨ.

2. н•„мҲҳ лӘЁл“Ҳ л°Ҹ мқҳмЎҙм„ұ м„Өм •

2.1 Python мһ„нҸ¬нҠё

TransformBroadcasterлҘј rclpy нҷҳкІҪм—җм„ң мӮ¬мҡ©н•ҳкё° мң„н•ҙм„ңлҠ” лӢӨмқҢмқҳ лӘЁл“Ҳмқ„ мһ„нҸ¬нҠён•ҳм—¬м•ј н•ңлӢӨ.

import rclpy
from rclpy.node import Node
from tf2_ros import TransformBroadcaster
from geometry_msgs.msg import TransformStamped

rclpyлҠ” ROS2 Python нҒҙлқјмқҙм–ёнҠё лқјмқҙлёҢлҹ¬лҰ¬мқҙл©°, tf2_ros.TransformBroadcasterлҠ” лҸҷм Ғ ліҖнҷҳ л°ңн–үмһҗ нҒҙлһҳмҠӨмқҙлӢӨ. geometry_msgs.msg.TransformStampedлҠ” мўҢн‘ң ліҖнҷҳ л©”мӢңм§Җмқҳ лҚ°мқҙн„° кө¬мЎ°лҘј м ңкіөн•ңлӢӨ.

мҝјн„°лӢҲм–ё ліҖнҷҳмқ„ мң„н•ҳм—¬ м¶”к°Җм ҒмңјлЎң лӢӨмқҢмқҳ лӘЁл“Ҳмқ„ мһ„нҸ¬нҠён•  мҲҳ мһҲлӢӨ.

from tf_transformations import quaternion_from_euler

tf_transformations нҢЁнӮӨм§Җк°Җ м„Өм№ҳлҗҳм–ҙ мһҲм§Җ м•ҠмқҖ кІҪмҡ°, scipyлҘј мӮ¬мҡ©н•ң лҢҖм•Ҳм Ғ л°©лІ•лҸ„ к°ҖлҠҘн•ҳлӢӨ.

from scipy.spatial.transform import Rotation

package.xml нҢҢмқјм—җ лӢӨмқҢмқҳ мқҳмЎҙм„ұмқ„ м„ м–ён•ҳм—¬м•ј н•ңлӢӨ.

Python нҢЁнӮӨм§Җмқҳ setup.pyм—җм„ң м§„мһ…м җ(entry point)мқ„ л“ұлЎқн•ңлӢӨ.

entry_points={
    'console_scripts': [

3. TransformBroadcaster нҒҙлһҳмҠӨмқҳ мқён„°нҺҳмқҙмҠӨ

3.1 нҒҙлһҳмҠӨ мғқм„ұ

tf2_ros.TransformBroadcasterлҠ” rclpy.node.Node мқёмҠӨн„ҙмҠӨлҘј мқёмһҗлЎң л°ӣм•„ мғқм„ұлҗңлӢӨ.

self.tf_broadcaster = TransformBroadcaster(self)

м—¬кё°м„ң selfлҠ” rclpy.node.NodeлҘј мғҒмҶҚн•ң нҒҙлһҳмҠӨмқҳ мқёмҠӨн„ҙмҠӨмқҙлӢӨ.

3.2 мЈјмҡ” л©”м„ңл“ң

TransformBroadcasterмқҳ н•өмӢ¬ л©”м„ңл“ңлҠ” sendTransform()мқҙлӢӨ. мқҙ л©”м„ңл“ңлҠ” лӢЁмқј TransformStamped л©”мӢңм§Җ лҳҗлҠ” TransformStamped л©”мӢңм§Җмқҳ лҰ¬мҠӨнҠёлҘј мқёмһҗлЎң л°ӣлҠ”лӢӨ.

# лӢЁмқј ліҖнҷҳ л°ңн–ү
self.tf_broadcaster.sendTransform(transform_stamped)

# ліөмҲҳ ліҖнҷҳ л°ңн–ү
self.tf_broadcaster.sendTransform([transform_stamped_1, transform_stamped_2])

лӢЁмқј ліҖнҷҳкіј ліөмҲҳ ліҖнҷҳ лӘЁл‘җ лҸҷмқјн•ң sendTransform() л©”м„ңл“ңлҘј нҶөн•ҳм—¬ мІҳлҰ¬лҗңлӢӨ. ліөмҲҳ ліҖнҷҳ л°ңн–ү мӢң лӘЁл“  ліҖнҷҳмқҖ н•ҳлӮҳмқҳ tf2_msgs/TFMessage л©”мӢңм§Җм—җ нҸ¬н•Ёлҗҳм–ҙ /tf нҶ н”ҪмңјлЎң м „мҶЎлҗңлӢӨ.

4. кё°ліё мӮ¬мҡ© нҢЁн„ҙ

4.1 л…ёл“ң нҒҙлһҳмҠӨ лӮҙл¶Җм—җм„ңмқҳ мҙҲкё°нҷ”

к°ҖмһҘ мқјл°ҳм Ғмқё мӮ¬мҡ© нҢЁн„ҙмқҖ rclpy.node.NodeлҘј мғҒмҶҚн•ң мӮ¬мҡ©мһҗ м •мқҳ нҒҙлһҳмҠӨ лӮҙл¶Җм—җм„ң TransformBroadcasterлҘј л©ӨлІ„ ліҖмҲҳлЎң м„ м–ён•ҳкі  мҙҲкё°нҷ”н•ҳлҠ” кІғмқҙлӢӨ.

import rclpy
from rclpy.node import Node
from tf2_ros import TransformBroadcaster
from geometry_msgs.msg import TransformStamped
from tf_transformations import quaternion_from_euler
import math


class DynamicTFBroadcaster(Node):
    def __init__(self):
        super().__init__('dynamic_tf_broadcaster')
        self.tf_broadcaster = TransformBroadcaster(self)
        self.timer = self.create_timer(0.1, self.broadcast_timer_callback)  # 10Hz

    def broadcast_timer_callback(self):
        t = TransformStamped()

        t.header.stamp = self.get_clock().now().to_msg()
        t.header.frame_id = 'world'
        t.child_frame_id = 'robot_base'

        # лі‘м§„ м„ұл¶„ м„Өм •
        t.transform.translation.x = 1.0
        t.transform.translation.y = 2.0
        t.transform.translation.z = 0.0

        # нҡҢм „ м„ұл¶„ м„Өм • (мҝјн„°лӢҲм–ё)
        q = quaternion_from_euler(0, 0, 1.57)  # Roll, Pitch, Yaw (лқјл””м•Ҳ)
        t.transform.rotation.x = q[0]
        t.transform.rotation.y = q[1]
        t.transform.rotation.z = q[2]
        t.transform.rotation.w = q[3]

        self.tf_broadcaster.sendTransform(t)

self.get_clock().now().to_msg()лҘј нҶөн•ҳм—¬ нҳ„мһ¬ ROS мӢңк°„мқ„ builtin_interfaces.msg.Time л©”мӢңм§Җ нҳ•мӢқмңјлЎң ліҖнҷҳн•ңлӢӨ. use_sim_time нҢҢлқјлҜён„°к°Җ нҷңм„ұнҷ”лҗң кІҪмҡ°, мӢңл®¬л Ҳмқҙм…ҳ мӢңк°„мқҙ л°ҳнҷҳлҗңлӢӨ.

4.2 main н•ЁмҲҳ кө¬м„ұ

def main(args=None):
    rclpy.init(args=args)
    node = DynamicTFBroadcaster()
    try:
        rclpy.spin(node)
    except KeyboardInterrupt:
        pass
    finally:
        node.destroy_node()
        rclpy.shutdown()


if __name__ == '__main__':
    main()

rclpy.spin()мқ„ нҶөн•ҳм—¬ л…ёл“ңлҘј мӢӨн–үн•ҳл©ҙ, нғҖмқҙлЁё мҪңл°ұмқҙ мЈјкё°м ҒмңјлЎң нҳём¶ңлҗҳм–ҙ мўҢн‘ң ліҖнҷҳмқ„ л°ңн–үн•ңлӢӨ. KeyboardInterrupt мҳҲмҷёлҘј мІҳлҰ¬н•ҳм—¬ Ctrl+Cм—җ мқҳн•ң м•Ҳм „н•ң мў…лЈҢлҘј ліҙмһҘн•ңлӢӨ.

5. TransformStamped л©”мӢңм§Җ кө¬м„ұ

5.1 н—ӨлҚ” (Header) м„Өм •

TransformStamped л©”мӢңм§Җмқҳ header н•„л“ңлҠ” ліҖнҷҳмқҳ мӢңк°„ м •ліҙмҷҖ л¶ҖлӘЁ н”„л Ҳмһ„ мӢқлі„мһҗлҘј нҸ¬н•Ён•ңлӢӨ.

t = TransformStamped()
t.header.stamp = self.get_clock().now().to_msg()
t.header.frame_id = 'parent_frame'

stamp н•„л“ңм—җлҠ” ліҖнҷҳмқҙ мң нҡЁн•ң мӢңк°Ғмқ„ м„Өм •н•ңлӢӨ. Pythonм—җм„ңлҠ” self.get_clock().now()к°Җ rclpy.time.Time к°қмІҙлҘј л°ҳнҷҳн•ҳлҜҖлЎң, .to_msg() л©”м„ңл“ңлҘј нҳём¶ңн•ҳм—¬ л©”мӢңм§Җ нҳ•мӢқмңјлЎң ліҖнҷҳн•ҳм—¬м•ј н•ңлӢӨ.

5.2 н”„л Ҳмһ„ мӢқлі„мһҗ м„Өм •

t.header.frame_id = 'world'       # л¶ҖлӘЁ н”„л Ҳмһ„
t.child_frame_id = 'sensor_link'  # мһҗмӢқ н”„л Ҳмһ„

frame_idлҠ” л¶ҖлӘЁ н”„л Ҳмһ„(parent frame)мқ„, child_frame_idлҠ” мһҗмӢқ н”„л Ҳмһ„(child frame)мқ„ м§Җм •н•ңлӢӨ. мқҙ л‘җ н•„л“ңлҠ” ліҖнҷҳ нҠёлҰ¬(transform tree)м—җм„ң н”„л Ҳмһ„ к°„ кі„мёө кҙҖкі„лҘј м •мқҳн•ңлӢӨ.

5.3 лі‘м§„ (Translation) м„Өм •

t.transform.translation.x = 0.5  # лҜён„° лӢЁмң„
t.transform.translation.y = 0.0
t.transform.translation.z = 0.3

лі‘м§„ лІЎн„°лҠ” л¶ҖлӘЁ н”„л Ҳмһ„ мӣҗм җм—җм„ң мһҗмӢқ н”„л Ҳмһ„ мӣҗм җк№Ңм§Җмқҳ ліҖмң„лҘј лӮҳнғҖлӮҙл©°, лӢЁмң„лҠ” лҜён„°(m)мқҙлӢӨ.

5.4 нҡҢм „ (Rotation) м„Өм •

нҡҢм „мқҖ мҝјн„°лӢҲм–ё(quaternion)мңјлЎң н‘ңнҳ„н•ңлӢӨ. tf_transformations нҢЁнӮӨм§Җмқҳ quaternion_from_euler() н•ЁмҲҳлҘј нҷңмҡ©н•ҳл©ҙ мҳӨмқјлҹ¬ к°Ғ(Euler angle)мңјлЎңл¶Җн„° мҝјн„°лӢҲм–ёмқ„ нҺёлҰ¬н•ҳкІҢ мғқм„ұн•  мҲҳ мһҲлӢӨ.

from tf_transformations import quaternion_from_euler

q = quaternion_from_euler(roll, pitch, yaw)  # лқјл””м•Ҳ лӢЁмң„
t.transform.rotation.x = q[0]
t.transform.rotation.y = q[1]
t.transform.rotation.z = q[2]
t.transform.rotation.w = q[3]

quaternion_from_euler() н•ЁмҲҳлҠ” Roll-Pitch-Yaw мҲңм„ңмқҳ мҳӨмқјлҹ¬ к°Ғмқ„ мһ…л ҘмңјлЎң л°ӣм•„ [x, y, z, w] мҲңм„ңмқҳ мҝјн„°лӢҲм–ё л°°м—ҙмқ„ л°ҳнҷҳн•ңлӢӨ. мқёмһҗмқҳ лӢЁмң„лҠ” лқјл””м•Ҳ(radian)мқҙлӢӨ.

5.4.1 scipyлҘј мӮ¬мҡ©н•ң лҢҖм•Ҳм Ғ л°©лІ•

tf_transformations нҢЁнӮӨм§Җк°Җ мӮ¬мҡ© л¶Ҳк°ҖлҠҘн•ң нҷҳкІҪм—җм„ңлҠ” scipyлҘј нҷңмҡ©н•  мҲҳ мһҲлӢӨ.

from scipy.spatial.transform import Rotation

r = Rotation.from_euler('xyz', [roll, pitch, yaw], degrees=False)
q = r.as_quat()  # [x, y, z, w] мҲңм„ң
t.transform.rotation.x = q[0]
t.transform.rotation.y = q[1]
t.transform.rotation.z = q[2]
t.transform.rotation.w = q[3]

нҡҢм „мқҙ м—ҶлҠ” кІҪмҡ°м—җлҠ” н•ӯл“ұ мҝјн„°лӢҲм–ё(identity quaternion)мқ„ м„Өм •н•ңлӢӨ.

t.transform.rotation.x = 0.0
t.transform.rotation.y = 0.0
t.transform.rotation.z = 0.0
t.transform.rotation.w = 1.0

6. ліөмҲҳ ліҖнҷҳмқҳ лҸҷмӢң л°ңн–ү

н•ҳлӮҳмқҳ л…ёл“ңм—җм„ң м—¬лҹ¬ мўҢн‘ң ліҖнҷҳмқ„ лҸҷмӢңм—җ л°ңн–үн•ҳм—¬м•ј н•ҳлҠ” кІҪмҡ°, лҰ¬мҠӨнҠёлҘј мӮ¬мҡ©н•ҳм—¬ ліөмҲҳмқҳ ліҖнҷҳмқ„ мқјкҙ„ м „мҶЎн•  мҲҳ мһҲлӢӨ.

def broadcast_multiple_transforms(self):
    transforms = []

    t1 = TransformStamped()
    t1.header.stamp = self.get_clock().now().to_msg()
    t1.header.frame_id = 'base_link'
    t1.child_frame_id = 'left_wheel'
    t1.transform.translation.x = 0.0
    t1.transform.translation.y = 0.15
    t1.transform.translation.z = 0.0
    t1.transform.rotation.w = 1.0
    transforms.append(t1)

    t2 = TransformStamped()
    t2.header.stamp = self.get_clock().now().to_msg()
    t2.header.frame_id = 'base_link'
    t2.child_frame_id = 'right_wheel'
    t2.transform.translation.x = 0.0
    t2.transform.translation.y = -0.15
    t2.transform.translation.z = 0.0
    t2.transform.rotation.w = 1.0
    transforms.append(t2)

    self.tf_broadcaster.sendTransform(transforms)

мқҙ л°©мӢқмқҖ л„ӨнҠёмӣҢнҒ¬ нҡЁмңЁм„ұ мёЎл©ҙм—җм„ң мң лҰ¬н•ҳлӢӨ. ліөмҲҳмқҳ ліҖнҷҳмқҙ н•ҳлӮҳмқҳ TFMessage л©”мӢңм§Җм—җ нҸ¬н•ЁлҗҳлҜҖлЎң, к°ңлі„м ҒмңјлЎң л°ңн–үн•ҳлҠ” кІғм—җ л№„н•ҳм—¬ нҶөмӢ  мҳӨлІ„н—Өл“ңк°Җ к°җмҶҢн•ңлӢӨ.

7. кө¬лҸ…мһҗ мҪңл°ұм—җм„ңмқҳ ліҖнҷҳ л°ңн–ү

мҷёл¶Җ м„јм„ң лҚ°мқҙн„°лӮҳ нҶ н”Ҫ л©”мӢңм§ҖлҘј мҲҳмӢ н•ң нӣ„, н•ҙлӢ№ лҚ°мқҙн„°лҘј кё°л°ҳмңјлЎң мўҢн‘ң ліҖнҷҳмқ„ л°ңн–үн•ҳлҠ” нҢЁн„ҙмқҙ л№ҲлІҲн•ҳкІҢ мӮ¬мҡ©лҗңлӢӨ.

from nav_msgs.msg import Odometry


class OdomTFBroadcaster(Node):
    def __init__(self):
        super().__init__('odom_tf_broadcaster')
        self.tf_broadcaster = TransformBroadcaster(self)
        self.odom_sub = self.create_subscription(
            Odometry,
            'odom',
            self.odom_callback,
            10
        )

    def odom_callback(self, msg):
        t = TransformStamped()

        t.header.stamp = msg.header.stamp
        t.header.frame_id = 'odom'
        t.child_frame_id = 'base_link'

        t.transform.translation.x = msg.pose.pose.position.x
        t.transform.translation.y = msg.pose.pose.position.y
        t.transform.translation.z = msg.pose.pose.position.z
        t.transform.rotation = msg.pose.pose.orientation

        self.tf_broadcaster.sendTransform(t)

мқҙ мҳҲм ңм—җм„ңлҠ” мҳӨлҸ„л©”нҠёлҰ¬(odometry) л©”мӢңм§ҖлҘј мҲҳмӢ н•  л•Ңл§ҲлӢӨ odom н”„л Ҳмһ„м—җм„ң base_link н”„л Ҳмһ„мңјлЎңмқҳ лҸҷм Ғ ліҖнҷҳмқ„ л°ңн–үн•ңлӢӨ. ліҖнҷҳмқҳ нғҖмһ„мҠӨнғ¬н”„лҠ” мҲҳмӢ н•ң л©”мӢңм§Җмқҳ нғҖмһ„мҠӨнғ¬н”„лҘј к·ёлҢҖлЎң мӮ¬мҡ©н•ҳм—¬ мӢңк°„м Ғ мқјкҙҖм„ұмқ„ ліҙмһҘн•ңлӢӨ.

8. нҢҢлқјлҜён„°лҘј нҷңмҡ©н•ң лҸҷм Ғ кө¬м„ұ

rclpyмқҳ нҢҢлқјлҜён„°(parameter) л©”м»ӨлӢҲмҰҳмқ„ нҷңмҡ©н•ҳл©ҙ, ліҖнҷҳ л°ңн–үмқҳ лҸҷмһ‘мқ„ лҹ°нғҖмһ„м—җ лҸҷм ҒмңјлЎң кө¬м„ұн•  мҲҳ мһҲлӢӨ.

class ParameterizedTFBroadcaster(Node):
    def __init__(self):
        super().__init__('parameterized_tf_broadcaster')

        # нҢҢлқјлҜён„° м„ м–ё
        self.declare_parameter('parent_frame', 'world')
        self.declare_parameter('child_frame', 'robot_base')
        self.declare_parameter('publish_frequency', 10.0)
        self.declare_parameter('x_offset', 0.0)
        self.declare_parameter('y_offset', 0.0)
        self.declare_parameter('z_offset', 0.0)

        self.tf_broadcaster = TransformBroadcaster(self)

        frequency = self.get_parameter('publish_frequency').value
        period = 1.0 / frequency
        self.timer = self.create_timer(period, self.timer_callback)

    def timer_callback(self):
        t = TransformStamped()
        t.header.stamp = self.get_clock().now().to_msg()
        t.header.frame_id = self.get_parameter('parent_frame').value
        t.child_frame_id = self.get_parameter('child_frame').value

        t.transform.translation.x = self.get_parameter('x_offset').value
        t.transform.translation.y = self.get_parameter('y_offset').value
        t.transform.translation.z = self.get_parameter('z_offset').value
        t.transform.rotation.w = 1.0

        self.tf_broadcaster.sendTransform(t)

import rclpy
from rclpy.node import Node
from tf2_ros import TransformBroadcaster
from geometry_msgs.msg import TransformStamped
from tf_transformations import quaternion_from_euler
import math


class RotatingFrameBroadcaster(Node):
    def __init__(self):
        super().__init__('rotating_frame_broadcaster')
        self.tf_broadcaster = TransformBroadcaster(self)
        self.timer = self.create_timer(0.05, self.timer_callback)  # 20Hz
        self.angle = 0.0
        self.get_logger().info('Rotating frame broadcaster started.')

    def timer_callback(self):
        t = TransformStamped()

        t.header.stamp = self.get_clock().now().to_msg()
        t.header.frame_id = 'world'
        t.child_frame_id = 'rotating_frame'

        # мӣҗнҳ• к¶Өм Ғ мғҒмқҳ лі‘м§„
        radius = 2.0
        t.transform.translation.x = radius * math.cos(self.angle)
        t.transform.translation.y = radius * math.sin(self.angle)
        t.transform.translation.z = 0.0

        # zм¶• нҡҢм „
        q = quaternion_from_euler(0, 0, self.angle)
        t.transform.rotation.x = q[0]
        t.transform.rotation.y = q[1]
        t.transform.rotation.z = q[2]
        t.transform.rotation.w = q[3]

        self.tf_broadcaster.sendTransform(t)

        self.angle += 0.02  # м•Ҫ 0.02 лқјл””м•Ҳ/н”„л Ҳмһ„
        if self.angle > 2.0 * math.pi:
            self.angle -= 2.0 * math.pi


def main(args=None):
    rclpy.init(args=args)
    node = RotatingFrameBroadcaster()
    try:
        rclpy.spin(node)
    except KeyboardInterrupt:
        pass
    finally:
        node.destroy_node()
        rclpy.shutdown()


if __name__ == '__main__':
    main()
